   📄 FAYL: .github/workflows/snake.yml
   (ilon animatsiyasi uchun)

name: Generate Snake Animation

on:
  schedule:
    - cron: "0 */12 * * *"   # har 12 soatda yangilanadi
  workflow_dispatch:           # qo'lda ishga tushirish mumkin
  push:
    branches:
      - main

jobs:
  generate:
    permissions:
      contents: write
    runs-on: ubuntu-latest
    timeout-minutes: 10

    steps:
      - name: Generate github-contribution-grid-snake.svg
        uses: Platane/snk/svg-only@v3
        with:
          github_user_name: ${{ github.repository_owner }}
          outputs: |
            dist/github-snake.svg
            dist/github-snake-dark.svg?palette=github-dark

      - name: Push snake animation to output branch
        uses: crazy-max/ghaction-github-pages@v3.1.0
        with:
          target_branch: output
          build_dir: dist
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}

────────────────────────────────────────────────────────────────
   📄 FAYL: .github/workflows/update-readme.yml
   (README ni avtomatik yangilash — ixtiyoriy)
────────────────────────────────────────────────────────────────

name: Update README

on:
  schedule:
    - cron: "0 0 * * *"    # har kecha yarim tunda
  workflow_dispatch:

jobs:
  update:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - name: Update README with latest stats
        run: echo "README yangilandi - $(date)" >> update.log
      - name: Commit changes
        run: |
          git config --local user.email "action@github.com"
          git config --local user.name "GitHub Action"
          git add -A
          git diff --quiet && git diff --staged --quiet || git commit -m "🤖 Auto-update README stats"
          git push
